\exercise{3.1.1.8}
\paragraph{a.}
\begin{align*}
  (a,b,c)\mapsto(a+b,c); (x,y)\mapsto xy
  &= (a,b,c)\mapsto ac+bc \\
  (a,b,c)\mapsto(a\cdot b,a\cdot c); (x,y)\mapsto x+y
  &= (a,b,c)\mapsto ab+ac
\end{align*}
does not commute
\paragraph{b.}
\begin{align*}
  x\mapsto(x,0); (a,b)\mapsto(a\cdot b) &= x\mapsto 0 \\
  id_\mathbb{Z} &= x\mapsto x
\end{align*}
does not commute
\paragraph{c.}
\begin{align*}
  x\mapsto(x,1); (a,b)\mapsto(a\cdot b) &= x\mapsto x \\
  id_\mathbb{Z} &= x\mapsto x
\end{align*}
commutes
